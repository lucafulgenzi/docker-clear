## Usage 
    git clone https://github.com/lucafulgenzi/docker-clear.git
    cd docker-clear
    bash docker-clear.sh
    
## Create alias
Edit .bashrc
   
    nano ~/.bashrc
    alias docker-clear="bash docker-clear.sh"
    source ~/.bashrc
